# -----------------------------------------------------------------------------

# Win32 API bindings:

#   win32service, win32serviceutil,
#   servicemanager, win32event              — main.py (Windows service framework)

#   python Scripts/pywin32_postinstall.py -install
pywin32==308

# WMI (Windows Management Instrumentation)
# Used by: security_software_monitor (antivirus, firewall, Defender status)
#          peripherals_monitor (display, audio fallback)
#          usb_devices_monitor (USB device enumeration)
#          specs_monitor (hardware specs via Get-CimInstance)
# Depends on pywin32 being installed and post-install script having been run
WMI==1.5.1


# -----------------------------------------------------------------------------
# MongoDB — optional direct-write fallback (AGENT_WRITE_TO_DB=true)
# -----------------------------------------------------------------------------

# Used by: base_monitor (fallback storage if backend API is unavailable)
# Optional — agent works without it; data goes to the backend API instead
pymongo==4.10.1


# -----------------------------------------------------------------------------
# GPU monitoring — optional, hardware_monitor only
# -----------------------------------------------------------------------------

# Primary NVIDIA GPU metrics (direct driver access, most accurate)
# Used by: hardware_monitor (_gpu_via_pynvml)
# Optional — omit if no NVIDIA GPU or GPU monitoring not needed
nvidia-ml-py==12.560.30     # pip name for pynvml

# Fallback NVIDIA GPU metrics (wraps nvidia-smi)
# Used by: hardware_monitor (_gpu_via_gputil) if pynvml unavailable
# Optional — only needed if pynvml is not installed
GPUtil==1.4.0


# -----------------------------------------------------------------------------
# Audio device enumeration — peripherals_monitor only
# -----------------------------------------------------------------------------

# Primary: Windows MMDevice API (most accurate, no duplicate devices)
# Used by: peripherals_monitor (_get_audio_pycaw)
# Optional — monitor degrades to sounddevice or WMI fallback without it

# Secondary fallback audio enumeration (PortAudio wrapper)
